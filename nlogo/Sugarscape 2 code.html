;; Sugarscape Geography Model
;; Adapted from Sugarscape 2 Constant Growback
;; Investigates how the spatial distribution of resource hills
;; affects survivability and inequality.

globals [
  gini-index-reserve
  lorenz-points
  total-sugar-budget
]

turtles-own [
  sugar
  metabolism
  vision
  vision-points
]

patches-own [
  psugar
  max-psugar
]

;;
;; Setup Procedures
;;

to setup
  clear-all
  setup-patches
  create-turtles initial-population [ turtle-setup ]
  update-lorenz-and-gini
  reset-ticks
end

to turtle-setup ;; turtle procedure
  set color red
  set shape "circle"
  move-to one-of patches with [not any? other turtles-here]
  set sugar random-in-range 10 20
  set metabolism random-in-range 1 3
  set vision random-in-range 1 3
  set vision-points []
  foreach (range 1 (vision + 1)) [ n ->
    set vision-points sentence vision-points (list (list 0 n) (list n 0) (list 0 (- n)) (list (- n) 0))
  ]
end

to setup-patches
  let hill-centers get-hill-centers

  let sugar-per-hill 4000 / num-hills

  ;; Correction factors to compensate for grid discretization
  let correction 1.0
  if num-hills = 1 [ set correction 0.925 ]
  if num-hills = 2 [ set correction 1.099 ]
  if num-hills = 3 [ set correction 1.022 ]
  if num-hills = 4 [ set correction 0.995 ]

  let w sqrt (sugar-per-hill * correction / (30 * 3.14159))

  ask patches [
    let min-dist 9999
    foreach hill-centers [ center-coords ->
      let cx item 0 center-coords
      let cy item 1 center-coords
      let d distancexy cx cy
      if d < min-dist [ set min-dist d ]
    ]
    ifelse min-dist <= w [ set max-psugar 4 ] [
      ifelse min-dist <= 2 * w [ set max-psugar 3 ] [
        ifelse min-dist <= 3 * w [ set max-psugar 2 ] [
          ifelse min-dist <= 4 * w [ set max-psugar 1 ] [
            set max-psugar 0
          ]
        ]
      ]
    ]
    set psugar max-psugar
    patch-recolor
  ]

  set total-sugar-budget sum [max-psugar] of patches
end

to-report get-hill-centers
  if num-hills = 1 [
    report (list (list 25 25))
  ]
  if num-hills = 2 [
    report (list (list 15 25) (list 35 25))
  ]
  if num-hills = 3 [
    report (list (list 25 35) (list 15 15) (list 35 15))
  ]
  ;; num-hills = 4 or fallback
  report (list (list 15 15) (list 35 15) (list 15 35) (list 35 35))
end

;;
;; Runtime Procedures
;;

to go
  if not any? turtles [ stop ]
  ask patches [
    patch-growback
    patch-recolor
  ]
  ask turtles [
    turtle-move
    turtle-eat
    if sugar <= 0 [ die ]
  ]
  update-lorenz-and-gini
  tick
end

to turtle-move
  let move-candidates (patch-set patch-here (patches at-points vision-points) with [not any? turtles-here])
  let possible-winners move-candidates with-max [psugar]
  if any? possible-winners [
    move-to min-one-of possible-winners [distance myself]
  ]
end

to turtle-eat
  set sugar (sugar - metabolism + psugar)
  set psugar 0
end

to patch-recolor
  set pcolor (yellow + 4.9 - psugar)
end

to patch-growback
  set psugar min (list max-psugar (psugar + 1))
end

;;
;; Lorenz Curve and Gini Coefficient
;;

to update-lorenz-and-gini
  let num-people count turtles
  if num-people = 0 [ set gini-index-reserve 0 set lorenz-points [] stop ]
  let sorted-wealths sort [sugar] of turtles
  let total-wealth sum sorted-wealths
  if total-wealth = 0 [ set gini-index-reserve 0 set lorenz-points [] stop ]
  let wealth-sum-so-far 0
  let index 0
  set gini-index-reserve 0
  set lorenz-points []
  repeat num-people [
    set wealth-sum-so-far (wealth-sum-so-far + item index sorted-wealths)
    set lorenz-points lput ((wealth-sum-so-far / total-wealth) * 100) lorenz-points
    set index (index + 1)
    set gini-index-reserve
      gini-index-reserve +
      (index / num-people) -
      (wealth-sum-so-far / total-wealth)
  ]
end

;;
;; Reporters for BehaviorSpace
;;

to-report gini
  report (gini-index-reserve / count turtles) * 2
end

to-report population
  report count turtles
end

to-report mean-wealth
  ifelse any? turtles [ report mean [sugar] of turtles ] [ report 0 ]
end

to-report median-wealth
  ifelse any? turtles [ report median [sugar] of turtles ] [ report 0 ]
end

to-report mean-vision
  ifelse any? turtles [ report mean [vision] of turtles ] [ report 0 ]
end

to-report mean-metabolism
  ifelse any? turtles [ report mean [metabolism] of turtles ] [ report 0 ]
end

to-report wealth-sd
  ifelse count turtles > 1 [ report standard-deviation [sugar] of turtles ] [ report 0 ]
end

to-report survival-rate
  report count turtles / initial-population
end

;;
;; Utilities
;;

to-report random-in-range [low high]
  report low + random (high - low + 1)
end
